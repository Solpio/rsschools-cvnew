Human Readble Numbers

Task: Create implement function toReadable that converts given number, to readable string.

                    
module.exports = function toReadable (number) {
    const namesOfNums = {
        0: "zero", 1: "one", 2: "two", 3: "three",
        4: "four", 5: "five", 6: "six", 7: "seven",
        8: "eight", 9: "nine", 10: "ten", 11: "eleven",
        12: "twelve", 13: "thirteen", 14: "fourteen",
        15: "fifteen", 16: "sixteen", 17: "seventeen", 
        18: "eighteen", 19: "nineteen",20: "twenty", 
        30: "thirty", 40: "forty", 50: "fifty", 
        60: "sixty", 70: "seventy", 80: "eighty", 
        90: "ninety",
    }
    const indexes = {
        0: "",
        1: "thousand",
        2: "million",
        3: "billion",
        4: "trillion",
        5: "quadrillion"
    }

    function _returnThreeNums(num) {
        let masThreeNums = [];
        while (num.toString().length > 3) {
            masThreeNums.push(Math.trunc(num % 1000));
            num = Math.trunc(num / 1000);
        }
        masThreeNums.push(Math.trunc(num % 1000));
        return masThreeNums;
    }

    function _decoder(mas) {
        let str = "";
        for (let nums in mas) {
            let threeStr = "";
            let hundred = Math.trunc(mas[nums] / 100);
            let dozen = Math.floor((mas[nums] % 100) / 10);
            let unit = mas[nums] % 10;
            if (hundred > 0) {
                threeStr = namesOfNums[hundred] + " hundred ";
            }
            if (dozen == 1) {
                threeStr = threeStr + namesOfNums[dozen * 10 + unit]
            } else {
                { if (dozen > 1){
                    threeStr = threeStr + namesOfNums[dozen * 10] + " ";
                }
                    if (unit > 0) {
                        threeStr = threeStr + namesOfNums[unit];
                    }
                }
            }
            if (unit == 0 && mas[nums].toString().length === 1) {
                threeStr = namesOfNums[unit];
            }
            if (nums > 0)
                threeStr = threeStr + ` ${indexes[nums]} `;

            str = threeStr + str;
        }
        return str.trim();
    }
    return _decoder(_returnThreeNums(number));
}
